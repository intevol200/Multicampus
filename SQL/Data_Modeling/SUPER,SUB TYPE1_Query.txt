SELECT
    c.cust_id,
    c.cust_name,
    c.jumin_num,
    o.order_id,
    o.order_date,
    o.order_total
FROM
    customers c,
    orders o
WHERE
    c.cust_id = o.cust_id
    AND   c.cust_type = 2;
